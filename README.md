- 함수 f에, x에 3, y에 6의 값이 들어가서, z = x(3)+y(6) = 9 가 되고, return z로 9를 출력한다. 

function f(x, y) {

        var z;
        
        z = x + y; // z = 3+6 = 9
        
        return z;
        
      }
      
      document.write(f(3, 6)); // 3 과 6이 위에 x,y 값으로 들어가서 계산된다.
